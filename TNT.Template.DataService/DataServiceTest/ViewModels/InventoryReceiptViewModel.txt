using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

		[JsonProperty("inventory_receipt_items")]
		public IEnumerable<InventoryReceiptItemViewModel> InventoryReceiptItemsVM { get; set; }
		
		public InventoryReceiptViewModel(InventoryReceipt entity) : this()
		{
			FromEntity(entity);
